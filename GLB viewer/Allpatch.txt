+export const DEFAULT_SETTINGS = {
+  shortcutsEnabled: true,
+  snapEnabled: true,
+  gridMm: 10,
+  angleDeg: 15,

+
+export function loadSettings() {
+  try {
+    const raw = localStorage.getItem(STORAGE_KEY);
+    if (!raw) return structuredClone(DEFAULT_SETTINGS);
+    return deepMerge(structuredClone(DEFAULT_SETTINGS), JSON.parse(raw));
+  } catch (err) {
+    console.warn('[ProGLB] settings load failed:', err);
+    return structuredClone(DEFAULT_SETTINGS);
+  }
+}
+
+export function saveSettings(next) {
+  try {
+    localStorage.setItem(STORAGE_KEY, JSON.stringify(next));
+  } catch (err) {
+    console.warn('[ProGLB] settings save failed:', err);
+  }
+}
+
+export function resetSettings() {
+  localStorage.removeItem(STORAGE_KEY);
+  return structuredClone(DEFAULT_SETTINGS);
+}
+
+function deepMerge(target, source) {
+  for (const key of Object.keys(source || {})) {
+    const value = source[key];
+    if (value && typeof value === 'object' && !Array.isArray(value)) {
+      target[key] = deepMerge(target[key] || {}, value);
+    } else {
+      target[key] = value;
+    }
+  }
+  return target;
+}

+export function createShortcutController({ onAction, enabled = true }) {
+  const keymap = new Map([
+    ['1', 'tool:select'],
+    ['2', 'tool:move'],
+    ['3', 'tool:rotate'],
+    ['w', 'tool:move'],
+    ['e', 'tool:rotate'],
+    ['f', 'view:fit'],
+    ['h', 'view:home'],
+    ['i', 'view:iso'],
+    ['t', 'view:top'],
+    ['n', 'view:front'],
+    ['g', 'toggle:snap'],
+    ['escape', 'tool:cancel'],
+    ['delete', 'selection:delete'],
+  ]);
+
+  function onKeyDown(event) {
+    if (!enabled) return;
+    if (shouldIgnore(event.target)) return;
+
+    const key = String(event.key || '').toLowerCase();
+
+    if ((event.ctrlKey || event.metaKey) && key === 'z' && !event.shiftKey) {
+      event.preventDefault();
+      onAction?.('history:undo', event);
+      return;
+    }
+    if ((event.ctrlKey || event.metaKey) && key === 'z' && event.shiftKey) {
+      event.preventDefault();
+      onAction?.('history:redo', event);
+      return;
+    }
+
+    const action = keymap.get(key);
+    if (!action) return;
+
+    event.preventDefault();
+    onAction?.(action, event);
+  }
+
+  window.addEventListener('keydown', onKeyDown);
+
+  return {
+    setEnabled(value) {
+      enabled = !!value;
+    },
+    destroy() {
+      window.removeEventListener('keydown', onKeyDown);
+    }
+  };
+}
+
+function shouldIgnore(target) {
+  if (!target) return false;
+  const tag = String(target.tagName || '').toLowerCase();
+  if (tag === 'input' || tag === 'textarea' || tag === 'select') return true;
+  if (target.isContentEditable) return true;
+  return false;
+}

This is the missing runtime glue for the UI layout patch. It gives your shell a real backend for:

* **loading a GLB**
* **fitting all geometry**
* **resetting home**
* **switching projection**
* **reading controller state**
* **hover ray-pick**
* **selection callbacks**
* **resize-safe rendering**
* **clean disposal**

Without this, the shell code is only a layout wrapper. With this, the shell becomes **usable**.

## Why this is the right companion

Your shell patch assumed these runtime methods existed. This companion patch exposes exactly those methods from `createViewerApp`, and makes `createSelection` and `createCameraController` projection-aware, which is necessary for:

* HUD updates
* hover tooltips
* aligned views
* ortho/perspective switching
* right-dock property display
* camera pass tests

That is fully consistent with the WI direction that the Pro GLB Viewer should reuse the advanced viewer modules but add structured shell, camera, property, and debug behavior on top.
